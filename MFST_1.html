def norm_dist_test(a):
    skew = sc.skew(a)
    skew_pvalue = sc.skewtest(a)[1]
    kurt = sc.kurtosis(a)
    kurt_pvalue = sc.kurtosistest(a)[1]
    norm_pvalue = sc.normaltest(a)[1]

    print(f' : {skew : .2f}')
    print(f' p-value : {skew_pvalue*100 : .2f}')
    print(f' : {kurt : .2f}')
    print(f' p-value : {kurt_pvalue*100 : .2f}')
    print(f'  p-value : {norm_pvalue*100 : .2f}')
